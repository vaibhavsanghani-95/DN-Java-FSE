Scenario-1:


BEGIN
    UPDATE Loans
    SET InterestRate = InterestRate - 1
    WHERE CustomerID IN (
        SELECT CustomerID
        FROM Customers
        WHERE Age > 60
    );

    DBMS_OUTPUT.PUT_LINE('Discount applied to eligible customers.');

    COMMIT;
END;
/


Scenario-2:

BEGIN
    UPDATE Customers
    SET IsVIP = 'TRUE'
    WHERE Balance > 10000;

    DBMS_OUTPUT.PUT_LINE('VIP status updated.');

    COMMIT;
END;
/


Scenario-3:

DECLARE
    CURSOR loan_cur IS
        SELECT l.LoanID,
               c.Name,
               l.DueDate
        FROM Loans l
        JOIN Customers c
        ON l.CustomerID = c.CustomerID
        WHERE l.DueDate BETWEEN SYSDATE AND SYSDATE + 30;
BEGIN
    FOR loan_rec IN loan_cur LOOP
        DBMS_OUTPUT.PUT_LINE(
            'Reminder: Loan ID ' || loan_rec.LoanID ||
            ' for customer ' || loan_rec.Name ||
            ' is due on ' || TO_CHAR(loan_rec.DueDate, 'DD-MON-YYYY')
        );
    END LOOP;
END;
/
